import streamlit as st

# Title of the calculator app
st.title("Simple Calculator App")

# Input fields for the user to enter numbers
num1 = st.number_input("Enter the first number:")
num2 = st.number_input("Enter the second number:")

# Select box to choose the operation
operation = st.selectbox("Select the operation", ["Addition", "Subtraction", "Multiplication", "Division"])

# Button to perform the calculation
if st.button("Calculate"):
    if operation == "Addition":
        result = num1 + num2
        st.success(f"The result of {num1} + {num2} is {result}")
    elif operation == "Subtraction":
        result = num1 - num2
        st.success(f"The result of {num1} - {num2} is {result}")
    elif operation == "Multiplication":
        result = num1 * num2
        st.success(f"The result of {num1} * {num2} is {result}")
    elif operation == "Division":
        if num2 == 0:
            st.error("Division by zero is not allowed")
        else:
            result = num1 / num2
            st.success(f"The result of {num1} / {num2} is {result}")
